[docs]    def readAverageVoltage(self, numavg, intertime):
        """Read the average voltage."""
        total = 0.0
        for _ in range(numavg):
            total += float(self._instrument.ask('MEASure?'))
            sleep(intertime)
        return (total/numavg, )
